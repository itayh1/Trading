pip:
1. Create new venv
2. install requirements.txt
 - install "numpy<2.0"
3. install: sudo apt install python3-pyqt5 python3-pyqt5.qtwebengine python3-pyqt5.qtwebchannel libqt5webkit5-dev
for ubuntu 22.04 (if doesn't work out of the box):
    - sudo apt install python3-gi python3-gi-cairo gir1.2-gtk-3.0 gir1.2-webkit2-4.1
    - sudo apt install libcairo2-dev libxt-dev libgirepository1.0-dev
    - sudo apt-get --reinstall install python3-gi
    - pip install PyGObject==3.50.0

conda:
1. Create conda environment
2. install conda requirements:
    - conda install "numpy<2.0"
    - conda install --file requirements.txt
    - pip install pandas_ta lightweight_charts

for linux do the following steps:
    - pip install pandas_ta lightweight_charts
    for running lightweight_charts:
        - sudo apt-get install build-essential libgl1-mesa-dev
        - pip3 install pyqt5 pyqtwebengine
        - pip3 install pywebview
        - pip3 install pywebview[qt]
